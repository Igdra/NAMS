59 const bool Thing::Move( Location* location )
60 {
61  UFLAGS_DE( flags );
62 
63  if ( location == NULL )
64  {
65  LOGSTR( flags, "Thing::Move()-> called with NULL location" );
66  return false;
67  }
68 
69  // If already stored inside another Thing then we need to remove ourselves first.
70  // If something prevents us from being removed, we can't move where we want to go.
71  if ( m_location != NULL )
72  {
73  if ( m_location->RemoveThing( this ) )
74  m_location = NULL;
75  else
76  return false;
77  }
78 
79  // Were we able to get into the destination Thing?
80  if ( location->AddThing( this ) )
81  m_location = location;
82  else
83  return false;
84 
85  return true;
86 }
