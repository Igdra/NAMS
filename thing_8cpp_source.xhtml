127 const void Thing::Send( const string& msg, Thing* speaker ) const
128 {
129  UFLAGS_DE( flags );

132 
133  if ( msg.empty() )

138 
139  for ( ti = m_contents.begin(); ti != m_contents.end(); ti++ )
140  {
141  thing = *ti;
