69 const bool Thing::Move( Thing* source, Thing* destination, Exit* exit )
70 {
71  UFLAGS_DE( flags );
72 
73  if ( source == NULL )
74  {
75  LOGSTR( flags, "Thing::Move()-> called with NULL source" );
76  return false;
77  }
78 
79  if ( destination == NULL )
80  {
81  LOGSTR( flags, "Thing::Move()-> called with NULL destination" );
82  return false;
83  }
84 
85  // If already stored inside another Thing then we need to remove ourselves first.
86  // If something prevents us from being removed, we can't move where we want to go.
87  if ( !source->RemoveThing( this ) )
88  return false;
89 
91  // Notify everyone else we moved
92  if ( exit )
93  source->Send( gName() + " leaves " + exit->gName() + "." CRLF, this );
94 
95  // Were we able to get into the destination Thing?
96  if ( !destination->AddThing( this ) )
97  return false;
98 
100  // Notify everyone else we arrived
101  if ( exit )
102  gContainer()->Send( gName() + " enters from " + exit->gName() + "." CRLF, this );
103 
104  return true;
105 }

142 const void Thing::Send( const string& msg, Thing* speaker, Thing* target ) const
143 {
144  UFLAGS_DE( flags );
145  CITER( vector, Thing*, ti );
146  Thing* thing = NULL;
147 
148  if ( msg.empty() )
149  {
150  LOGSTR( flags, "Thing::Send()-> called with empty msg" );
151  return;
152  }
153 
154  for ( ti = m_contents.begin(); ti != m_contents.end(); ti++ )
155  {
156  thing = *ti;
157 
158  if ( thing == speaker || thing == target )
159  continue;
160  else
161  thing->Send( msg );
162  }
163 
164  return;
165 }
