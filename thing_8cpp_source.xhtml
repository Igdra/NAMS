59 const bool Thing::Move( Location* location )
60 {

62 
63  if ( location == NULL )
64  {
65  LOGSTR( flags, "Thing::Move()-> called with NULL location" );
66  return false;
67  }
68 
69  // If already stored inside another Thing then we need to remove ourselves first.
70  // If something prevents us from being removed, we can't move where we want to go.
71  if ( m_location != NULL )

118 const void Thing::Send( const string& msg, Thing* speaker ) const
119 {
120  UFLAGS_DE( flags );
121  CITER( vector, Thing*, ti );
122  Thing* thing = NULL;
123 
124  if ( msg.empty() )
125  {
126  LOGSTR( flags, "Thing::Send()-> called with empty msg" );
127  return;
128  }
129 
130  for ( ti = m_contents.begin(); ti != m_contents.end(); ti++ )
131  {
132  thing = *ti;
133 
134  if ( thing == speaker )
135  continue;
136  else
137  thing->Send( msg );
138  }
139 
140  return;
141 }
