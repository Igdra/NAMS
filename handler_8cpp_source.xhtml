339 Thing* Handler::FindThing( const string& name, const uint_t& type, const uint_t& scope, Thing* caller, const bool& self )
340 {
341  UFLAGS_DE( flags );
342  bool found = false;
343  Thing* thing = NULL;
344  vector<Thing*> targets;
345  CITER( vector, Thing*, ti );
346  uint_t ltype = type, lscope = scope;
347 
348  if ( name.empty() )
349  {
350  LOGSTR( flags, "Handler::FindThing()-> called with empty name" );
351  return thing;
352  }
353 
354  if ( caller == NULL )
355  {
356  LOGSTR( flags, "Handler::FindThing()-> called with NULL caller" );
357  return thing;
358  }
359 
360  if ( ltype < uintmin_t || ltype >= MAX_THING_TYPE )
361  {
362  LOGFMT( flags, "Handler::FindThing()-> called with invalid type: %lu", ltype );
363  LOGSTR( flags, "Handler::FindThing()-> defaulting to THING_TYPE_THING" );
364  ltype = THING_TYPE_THING;
365  }
366 
367  if ( lscope < uintmin_t || lscope >= MAX_HANDLER_SCOPE )
368  {
369  LOGFMT( flags, "Handler::FindThing()-> called with invalid scope: %lu", lscope );
370  LOGSTR( flags, "Handler::FindThing()-> defaulting to HANDLER_SCOPE_LOC_INV" );
371  lscope = HANDLER_SCOPE_LOC_INV;
372  }
373 
374  switch ( lscope )
375  {
376  case HANDLER_SCOPE_INVENTORY:
377  targets = caller->gContents();
378  for ( ti = targets.begin(); ti != targets.end(); ti++ )
379  {
380  thing = *ti;
381 
382  // Shouldn't ever happen anyways...
383  if ( thing == caller && !self )
384  continue;
385 
386  if ( thing->gType() != ltype )
387  continue;
388 
389  if ( Utils::iName( name, thing->gName() ) )
390  {
391  found = true;
392  break;
393  }
394  }
395  break;
396 
397  case HANDLER_SCOPE_LOCATION:
398  if ( !caller->gContainer() )
399  {
400  LOGSTR( flags, "Handler::FindThing()->Thing::gContainer()-> returned NULL" );
401  return thing;
402  }
403 
404  targets = caller->gContainer()->gContents();
405  for ( ti = targets.begin(); ti != targets.end(); ti++ )
406  {
407  thing = *ti;
408 
409  // Shouldn't ever happen anyways...
410  if ( thing == caller && !self )
411  continue;
412 
413  if ( thing->gType() != ltype )
414  continue;
415 
416  if ( Utils::iName( name, thing->gName() ) )
417  {
418  found = true;
419  break;
420  }
421  }
422  break;
423 
424  case HANDLER_SCOPE_LOC_INV:
425  if ( ( thing = FindThing( name, ltype, HANDLER_SCOPE_LOCATION, caller ) ) == NULL )
426  thing = FindThing( name, ltype, HANDLER_SCOPE_INVENTORY, caller );
427 
428  if ( thing )

431  }
432 
433  if ( !found )
