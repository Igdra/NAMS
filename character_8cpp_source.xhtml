60 const void Character::Interpret( const uint_t& security, const string& cmd, const string& args )
61 {
62  Command* command = NULL;
63  Exit* exit = NULL;
64 
65  if ( ( command = Handler::FindCommand( cmd ) ) != NULL )
66  {
67  if ( command->Authorized( security ) )
68  command->Run( this, cmd, args );
69  else
70  Send( CFG_STR_CMD_INVALID );

72  else if ( ( exit = Handler::FindExit( gLocation(), cmd ) ) != NULL ) // Search for an exit

77  command->Run( this );
78  }
79  else // Give up
80  Send( CFG_STR_CMD_INVALID );

82  return;
83 }

173 const bool Character::Unserialize()
174 {
175  UFLAGS_DE( flags );
176  UFLAGS_I( finfo );
177  ifstream ifs;
178  string key, value, line;
179  stringstream loop, mline;
180  bool found = false, maxb = false;
181  uint_t revision = uintmin_t;
182 
183  Utils::FileOpen( ifs, Utils::DirPath( Utils::DirPath( CFG_DAT_DIR_ACCOUNT, gAccount()->gId() ), m_file ) );
184 
185  if ( !ifs.good() )
186  {
187  LOGFMT( flags, "Character::Unserialize()-> failed to open character file: %s", CSTR( m_file ) );
188  return false;
189  }
190 
191  while ( getline( ifs, line ) )
192  {
193  if ( !Utils::KeyValue( key, value, line) )
194  {
195  LOGFMT( flags, "Character::Unserialize()-> error reading line: %s", CSTR( line ) );
196  continue;
197  }
198 
199  for ( ;; )
200  {
201  found = false;
202  maxb = false;
203 
204  if ( Utils::StrPrefix( "description", key ) )
205  {
206  found = true;
207  sDescription( Utils::ReadString( ifs ), Utils::ReadIndex( key ) );
208  }
209  else if ( key == "id" )
210  {
211  found = true;
212  sId( value );
213  }
214  else if ( key == "location" )
215  {
216  found = true;
217  gAccount()->gClient()->sLogin( SOC_LOGIN_LOCATION, value );
218  }
219  else if ( key == "name" )
220  {
221  found = true;
222  sName( value );
223  }
224 
225  Utils::KeySet( true, found, key, "revision", value, revision, CFG_CHR_REVISION, maxb );
226  Utils::KeySet( true, found, key, "sex", value, m_sex, MAX_CHR_SEX, maxb );
227 
228  if ( !found )
229  LOGFMT( flags, "Character::Unserialize()-> key not found: %s", CSTR( key ) );
230 
231  if ( maxb )
232  LOGFMT( finfo, "Character::Unserialize()-> character id %s, key %s has illegal value %s", CSTR( gId() ), CSTR( key ), CSTR( value ) );
233 
234  break;
235  }
236  }
237 
238  Utils::FileClose( ifs );
239 
240  return true;
241 }
