177 const bool Object::Unserialize()
178 {
179  UFLAGS_DE( flags );
180  UFLAGS_I( finfo );
181  ifstream ifs;
182  string key, value, line;
183  stringstream loop, mline;
184  bool found = false, maxb = false;
185  uint_t revision = uintmin_t;
186 
187  Utils::FileOpen( ifs, m_file );
188 
189  if ( !ifs.good() )
190  {
191  LOGFMT( flags, "Object::Unserialize()-> failed to open object file: %s", CSTR( m_file ) );
192  return false;
193  }
194 
195  while ( getline( ifs, line ) )
196  {
197  if ( line.empty() )
198  getline( ifs, line );
199 
200  if ( !Utils::KeyValue( key, value, line) )
201  {
202  LOGFMT( flags, "Object::Unserialize()-> error reading line: %s", CSTR( line ) );
203  continue;
204  }
205 
206  for ( ;; )
207  {
208  found = false;
209  maxb = false;
210 
211  if ( Utils::StrPrefix( "description", key ) )
212  {
213  found = true;
214  sDescription( Utils::ReadString( ifs ), Utils::ReadIndex( key ) );
215  }
216  else if ( key == "id" )
217  {
218  found = true;
219  sId( value );
220  }
221  else if ( key == "name" )
222  {
223  found = true;
224  sName( value );
225  }
226  Utils::KeySet( true, found, key, "revision", value, revision, CFG_OBJ_REVISION, maxb );
227  Utils::KeySet( true, found, key, "zone", value, m_zone );
228 
229  if ( !found )
230  LOGFMT( flags, "Object::Unserialize()-> key not found: %s", CSTR( key ) );
231 
232  if ( maxb )
233  LOGFMT( finfo, "Object::Unserialize()-> object id %s, key %s has illegal value %s", CSTR( gId() ), CSTR( key ), CSTR( value ) );
234 
235  break;
236  }
237  }
238 
239  Utils::FileClose( ifs );
240 
241  return true;
242 }
