169 const bool SocketClient::ProcessCommand()
170 {
171  UFLAGS_DE( flags );
172  pair<string,string> cmd;
173  Command* command = NULL;
174  uint_t security = ACT_SECURITY_NONE;
175 
176  if ( !Valid() )
177  {
178  LOGFMT( flags, "SocketClient::ProcessCommand()->SocketClient::Valid()-> descriptor %ld returned false", gDescriptor() );
179  return false;
180  }
181 
182  // Nothing new to process; move along
183  if ( m_command_queue.empty() )
184  return true;
185 
186  if ( m_account )
187  security = m_account->gSecurity();
188 
189  cmd = m_command_queue.front();
190  m_command_queue.pop_front();
191 
192  // Redirect if not fully logged in yet
193  if ( m_state < SOC_STATE_PLAYING )
194  Handler::LoginHandler( this, cmd.first, cmd.second );
195  else
196  {
197  // Raw run if no Character exists to interpret
198  if ( m_account == NULL || m_account->gCharacter() == NULL )
199  {
200  if ( ( command = Handler::FindCommand( cmd.first ) ) != NULL )
201  {
202  if ( command->Authorized( security ) )
203  command->Run( this, cmd.first, cmd.second );
204  else
205  Send( CFG_STR_CMD_INVALID );
206  }
207  else
208  Send( CFG_STR_CMD_INVALID );
209  }
210  else
211  m_account->gCharacter()->Interpret( security, cmd.first, cmd.second );
212  }
213 
214  return true;
215 }
