61 const void Command::Delete()

94 const bool Command::New( const string& file )
95 {
96  UFLAGS_DE( flags );
97  string path( Utils::DirPath( CFG_DAT_DIR_OBJ, file, CFG_PLG_BUILD_EXT_OUT ) );
98  vector<string> disabled = g_config->gDisabledCommands();
99 
100  // Ensure there is a valid file to open
101  if ( !Utils::iFile( path ) )
102  {
103  LOGFMT( flags, "Command::New()->Utils::iFile()-> returned false for path: %s", CSTR( path ) );
104  return false;
105  }
106 
107  if ( ( m_plg_handle = ::dlopen( CSTR( path ), RTLD_LAZY | RTLD_GLOBAL ) ) == NULL )
108  {
109  LOGFMT( flags, "Command::New()->dlopen() returned error: %s", ::dlerror() );
110  return false;
111  }
112  else
113  {
114  // Register the handlers and create a new instance of the Plugin's class
115  m_plg_delete = (PluginDelete*) ::dlsym( m_plg_handle, "Delete" );
116  m_plg_file = file;
117  m_plg_new = (PluginNew*) ::dlsym( m_plg_handle, "New" );
118  m_plg = m_plg_new();
119  m_plg->sCaller( this );
120 
121  // Set specific values unique to a Command object that the Plugin can specify
122  m_preempt = m_plg->gBool( PLG_TYPE_COMMAND_BOOL_PREEMPT );
123  m_security = m_plg->gUint( PLG_TYPE_COMMAND_UINT_SECURITY );
124 
125  // Check to see if the Command has been disabled
126  if ( find( disabled.begin(), disabled.end(), gName() ) != disabled.end() )
127  m_disabled = true;
128  }
129 
130  if ( CFG_GAM_CMD_IGNORE_CASE )
131  command_list.insert( pair<const char,Command*>( Utils::Lower( gName() )[0], this ) );
132  else
133  command_list.insert( pair<const char,Command*>( gName()[0], this ) );
134 
135  return true;
136 }
