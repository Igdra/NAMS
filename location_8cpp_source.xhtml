153 const bool Location::Unserialize()
154 {
155  UFLAGS_DE( flags );
156  UFLAGS_I( finfo );
157  ifstream ifs;
158  string key, value, line;
159  stringstream loop, mline;
160  bool found = false, maxb = false;
161  uint_t revision = uintmin_t;
162 
163  Utils::FileOpen( ifs, m_file );
164 
165  if ( !ifs.good() )
166  {
167  LOGFMT( flags, "Location::Unserialize()-> failed to open location file: %s", CSTR( m_file ) );
168  return false;
169  }
170 
171  while ( getline( ifs, line ) )
172  {
173  if ( line.empty() )
174  getline( ifs, line );
175 
176  if ( !Utils::KeyValue( key, value, line) )
177  {
178  LOGFMT( flags, "Location::Unserialize()-> error reading line: %s", CSTR( line ) );
179  continue;
180  }
181 
182  for ( ;; )
183  {
184  found = false;
185  maxb = false;
186 
187  if ( Utils::StrPrefix( "description", key ) )
188  {
189  found = true;
190  sDescription( Utils::ReadString( ifs ), Utils::ReadIndex( key ) );
191  }
192  else if ( key == "exit" )
193  {
194  Exit* exit = NULL;
195 
196  found = true;
197  exit = new Exit();
198 
199  if ( !exit->New( this ) )
200  {
201  LOGSTR( flags, "Location::Unserialize()->Exit::New()-> returned false" );
202  exit->Delete();
203  }
204  else
205  {
206  if ( !exit->Unserialize( value ) )
207  {
208  LOGSTR( flags, "Location::Unserialize()->Exit::Unserialize()-> returned false" );
209  exit->Delete();
210  }
211  else
212  m_exits.push_back( exit );
213  }
214  }
215  else if ( key == "id" )
216  {
217  found = true;
218  sId( value );
219  }
220  else if ( key == "name" )
221  {
222  found = true;
223  sName( value );
224  }
225  Utils::KeySet( true, found, key, "revision", value, revision, CFG_LOC_REVISION, maxb );
226  Utils::KeySet( true, found, key, "zone", value, m_zone );
227 
228  if ( !found )
229  LOGFMT( flags, "Location::Unserialize()-> key not found: %s", CSTR( key ) );
230 
231  if ( maxb )
232  LOGFMT( finfo, "Location::Unserialize()-> location id %s, key %s has illegal value %s", CSTR( gId() ), CSTR( key ), CSTR( value ) );
233 
234  break;
235  }
236  }
237 
238  Utils::FileClose( ifs );
239 
240  return true;
241 }
