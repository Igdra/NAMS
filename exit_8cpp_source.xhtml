104  item = Utils::ReadPair( arg );

106  Utils::KeySet( true, found, item.first, "dest_id", item.second, m_dest_id );
107  Utils::KeySet( true, found, item.first, "name", item.second, m_name );
108 
109  if ( !found )

111  }
112 
113  return true;
